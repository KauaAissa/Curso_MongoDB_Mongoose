## Relacionamento One to One

db.pessoas.insertOne({
    nome:"Matheus",
    idade:30,
    profissao:"Programador"
})

const p = db.pessoas.findOne()

db.enderecos.insertOne({
    rua:"Rua das flores",
    numero:"8719",
    complemento:"Casa",
    pessoa_id: p._id
})

db.enderecos.find({})

db.enderecos.find({pessoa_id: p._id})

## Relacionamento One to Many

db.pessoas.insertOne({
    nome:"Ronaldo",
    idade:27,
    profissao:"Gerente"
})

const ronaldo = db.pessoas.findOne({nome:"Ronaldo"})

ronaldoId = ronaldo._id

db.compras.insertMany({
    {produtos: ["Livro", "Celular"], pessoa_id: ronaldoId},
    {produtos: ["Agenda"], pessoa_id: ronaldoId},
})


## Many to Many
db.cursos.insertMany([

])

db.curso_pessoa.insertMany([
    {curso_id: js._id, pessoa_id: matheus._id},
